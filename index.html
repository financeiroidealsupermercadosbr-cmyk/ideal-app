Apps Script
Para salvar baixas e despesas no Google Sheets, siga:

1. Abra a planilha do Google Sheets
2. Menu: Extensões → Apps Script
3. Apague o código existente e cole o código abaixo
4. Clique em Implantar → Nova implantação
5. Tipo: App da Web, acesso: Qualquer pessoa
6. Copie a URL gerada e cole no campo abaixo
function doPost(e) { var d = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.openById( '1ICENG-MlIKXuWhu5vOwEETf0xtdE_Myn45Exlbuqnw4'); var out = ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); if (d.acao === 'baixa') { var ws = ss.getSheetByName('Contas_Pagar'); var row = parseInt(d.linha); ws.getRange(row, 4).setValue(d.dataPag); ws.getRange(row, 11).setValue('PAGO'); ws.getRange(row, 12).setValue(d.banco); } if (d.acao === 'despesa') { var ws2 = ss.getSheetByName('Despesas_Caixa'); if (!ws2) ws2 = ss.insertSheet('Despesas_Caixa'); ws2.appendRow([d.data, d.loja, d.categoria, d.desc, d.valor, d.resp, new Date().toLocaleString('pt-BR')]); } return out; }
